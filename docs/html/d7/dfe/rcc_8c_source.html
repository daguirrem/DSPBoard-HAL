70void rcc_sysclk_select(rcc_sysclk_t source)
71{
72 uint32_t RCC_CR_BIT_RD = 0; /* Target ready bit on RCC_CR register */
73 uint32_t pll_request;
74 uint32_t cfgr_switch;
75 uint32_t cfgr_ready;
76
77 const uint32_t APB_PRESCALER[] = {
78 0x000,
79 0x100,
80 0x101,
81 0x110,
82 0x111
83 };
84
85 /* Separe PLL request on */
86 pll_request = source & RCC_CR_PLLON;
87 source &= ~RCC_CR_PLLON;
88
89 switch (source)
90 {
91 case RCC_SYSCLK_HSE:
92 RCC_CR_BIT_RD = RCC_CR_HSERDY;
93 cfgr_switch = 1;
94 break;
95 case RCC_SYSCLK_HSI:
96 default:
97 RCC_CR_BIT_RD = RCC_CR_HSIRDY;
98 cfgr_switch = 0;
99 break;
100 }
101 /* Power on the target clock (HSI or HSE) */
102 RCC->CR |= source;
103 /* Wait for stable target clock (rdy flag == 1) */
104 while ((RCC->CR & RCC_CR_BIT_RD) == 0);
105 /* Verify PLL target and power on it if required */
106 if(pll_request)
107 {
108 cfgr_switch = 2;
109 RCC->CR |= pll_request;
110 /* Wait for stable pll clock (rdy flag == 1) */
111 while ((RCC->CR & RCC_CR_PLLRDY) == 0);
112 /* Set maximus AHB/APB bus clocks */
113 RCC->CFGR &= ~((0x7 << RCC_CFGR_PPRE2_Pos) | (0x7 << RCC_CFGR_PPRE1_Pos)
114 | (0xF << RCC_CFGR_HPRE_Pos));
115 RCC->CFGR |= (APB_PRESCALER[pll_parameters.ppre2] << RCC_CFGR_PPRE2_Pos)
116 | (APB_PRESCALER[pll_parameters.ppre1] << RCC_CFGR_PPRE1_Pos);
117 }
118
119 /* Determinate the target system core clock to FLASH config */
120 switch (cfgr_switch)
121 {
122 case 0:
123 system_clk = HSI_VALUE;
124 break;
125 case 1:
126 system_clk = HSE_VALUE;
127 break;
128 case 2:
129 system_clk = pll_parameters.freq_o;
130 break;
131 default:
132 break;
133 }
134
135 /* Referer to Relation between CPU clock frequency and Flash memory read
136 time [increasing/decreasing the CPU Freq.] pag. 81 - Refrence Manual*/
137 if (system_clk > SystemCoreClock)
138 {
139 flash_acr_config_auto(system_clk);
140 }
141
142 cfgr_ready = (cfgr_switch<<RCC_CFGR_SWS_Pos);
143 /* Switch clock from system */
144 RCC->CFGR &= ~0x3; RCC->CFGR |= cfgr_switch;
145 /* Wait for the system complete the switch */
146 while ((RCC->CFGR & cfgr_ready) != cfgr_ready);
147
148 if (system_clk < SystemCoreClock)
149 {
150 flash_acr_config_auto(system_clk);
151 }
152
153 /* Update the internal monitor clock variable */
154 sys_clk_update();
155 /* Shutdown the other source clocks */
156 RCC->CR &= ((RCC_CR_PLLON | RCC_CR_HSEON| RCC_CR_HSION) & (source | pll_request));
157}

159void rcc_pll_param_cpte(uint8_t pll_clk_out, rcc_pllclk_t pll_clk_in)
160{
161 pll_parameters.source = pll_clk_in;
162 pll_parameters.freq_o = pll_clk_out * 1000000;
163
164 const uint8_t pll_p[] = /* Decode table for PLL_P values */
165 {
166 2,4,6,8
167 };
168
169 /* 2 MHz target fvco_in (suggested by reference manual) (pag. 227) */
170 /* input vco frequency [MHz] */
171 /* output vco frequency [MHz] */
172 const uint32_t fvco_in = 2;
173 uint32_t fvco_out;
174
175 /* Determinate PLL clock */
176 uint32_t __pll_clk_in = (pll_clk_in == RCC_PLLCLK_HSE) ? HSE_VALUE : HSI_VALUE;
177 __pll_clk_in /= 1000000UL;
178
179 /* Configure de max value to pll_p parameter */
180 pll_parameters.p = 3; /* Bit value for request pll_p */
181
182 /* Determinate the minimun pll_p for request clock */
183 do {
184 uint8_t idx = pll_parameters.p;
185 /* fvco_out = pll_out * pll_p (pag. 226 - reference manual) */
186 fvco_out = pll_clk_out * pll_p[idx];
187 if (fvco_out > RCC_PLLN_MAX_VALUE) {
188 /* If fvco_out is major than FVCO_MAX decrease pll_p
189 to get a valid value for fvco_out freq */
190 pll_parameters.p -= 1;
191 } else {
192 /* Otherwise break the while routine */
193 break;
194 }
195 } while(pll_parameters.p > 0);
196 /* Recalculate fvco with the last pll_p value */
197 fvco_out = pll_clk_out * pll_p[pll_parameters.p];
198 /* Compute other parameters */
199 pll_parameters.m = __pll_clk_in / fvco_in;
200 pll_parameters.n = fvco_out / fvco_in;
201 pll_parameters.q = fvco_out / 48;
202 /* PLL_Q Must not be less than 2 */
203 if (pll_parameters.q < 2)
204 {
205 pll_parameters.q = 2;
206 }
207 /* Compute APB preescalers to get max bus speeds */
208 pll_parameters.ppre1 = pll_clk_out / 42; /* 42MHz in APB1 case */
209 pll_parameters.ppre2 = pll_clk_out / 84; /* 84MHz in APB2 case */
210}
