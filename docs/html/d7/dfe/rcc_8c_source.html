111void rcc_sysclk_select(rcc_sysclk_t source)
112{
113 uint32_t RCC_CR_BIT_RD = 0; /* Target ready bit on RCC_CR register */
114 uint32_t pll_request;
115 uint32_t cfgr_switch;
116 uint32_t cfgr_ready;
117
118 const uint32_t APB_PRESCALER[] = {
119 0x000,
120 0x100,
121 0x101,
122 0x110,
123 0x111
124 };
125
126 /* Store clocks before change */
127 rcc_clocks_previous.system = rcc_clocks_current.system;
128 rcc_clocks_previous.ahb = rcc_clocks_current.ahb;
129
130 /* Separe PLL request on */
131 pll_request = source & RCC_CR_PLLON;
132 source &= ~RCC_CR_PLLON;
133
134 switch (source)
135 {
136 case RCC_SYSCLK_HSE:
137 RCC_CR_BIT_RD = RCC_CR_HSERDY;
138 cfgr_switch = 1;
139 break;
140 case RCC_SYSCLK_HSI:
141 default:
142 RCC_CR_BIT_RD = RCC_CR_HSIRDY;
143 cfgr_switch = 0;
144 break;
145 }
146 /* Power on the target clock (HSI or HSE) */
147 RCC->CR |= source;
148 /* Wait for stable target clock (rdy flag == 1) */
149 while ((RCC->CR & RCC_CR_BIT_RD) == 0);
150 /* Verify PLL target and power on it if required */
151 if(pll_request)
152 {
153 cfgr_switch = 2;
154 RCC->CR |= pll_request;
155 /* Wait for stable pll clock (rdy flag == 1) */
156 while ((RCC->CR & RCC_CR_PLLRDY) == 0);
157 /* Set maximus AHB/APB bus clocks */
158 RCC->CFGR &= ~((0x7 << RCC_CFGR_PPRE2_Pos) | (0x7 << RCC_CFGR_PPRE1_Pos)
159 | (0xF << RCC_CFGR_HPRE_Pos));
160 RCC->CFGR |= (APB_PRESCALER[pll_parameters.ppre2] << RCC_CFGR_PPRE2_Pos)
161 | (APB_PRESCALER[pll_parameters.ppre1] << RCC_CFGR_PPRE1_Pos);
162 }
163
164 /* Determinate the target system core clock to FLASH config */
165 uint32_t system_clk;
166 switch (cfgr_switch)
167 {
168 case 1:
169 system_clk = HSE_VALUE;
170 break;
171 case 2:
172 system_clk = pll_parameters.freq_o;
173 break;
174 case 0:
175 default:
176 system_clk = HSI_VALUE;
177 break;
178 }
179
180 /* Referer to Relation between CPU clock frequency and Flash memory read
181 time [increasing/decreasing the CPU Freq.] pag. 81 - Refrence Manual*/
182 if (system_clk > SystemCoreClock)
183 {
184 flash_acr_config_auto(system_clk);
185 }
186
187 cfgr_ready = (cfgr_switch<<RCC_CFGR_SWS_Pos);
188 /* Switch clock from system */
189 RCC->CFGR &= ~0x3; RCC->CFGR |= cfgr_switch;
190 /* Wait for the system complete the switch */
191 uint32_t tmp;
192 do {
193 tmp = (RCC->CFGR & 0xC);
194 }
195 while (tmp != cfgr_ready);
196
197 if (system_clk < SystemCoreClock)
198 {
199 flash_acr_config_auto(system_clk);
200 }
201
202 /* Update the internal monitor clock variable */
203 sys_clk_update();
204 /* Shutdown the other source clocks */
205 RCC->CR &= ((RCC_CR_PLLON | RCC_CR_HSEON| RCC_CR_HSION) & (source | pll_request));
206
207 /* Update current clocks */
208 rcc_clocks_current.system = system_clk;
209 rcc_clocks_current.ahb = system_clk;
210}

212void rcc_pll_param_cpte(uint8_t pll_clk_out, rcc_pllclk_t pll_clk_in)
213{
214 pll_parameters.source = pll_clk_in;
215 pll_parameters.freq_o = pll_clk_out * 1000000;
216
217 /* Save previous bus speeds */
218 rcc_clocks_previous.apb1 = rcc_clocks_current.apb1;
219 rcc_clocks_previous.apb2 = rcc_clocks_current.apb2;
220
221 const uint8_t pll_p[] = /* Decode table for PLL_P values */
222 {
223 2,4,6,8
224 };
225
226 /* 2 MHz target fvco_in (suggested by reference manual) (pag. 227) */
227 /* input vco frequency [MHz] */
228 const uint32_t fvco_in = 2;
229 /* output vco frequency [MHz] */
230 uint32_t fvco_out;
231
232 /* Determinate PLL clock */
233 uint32_t __pll_clk_in = (pll_clk_in == RCC_PLLCLK_HSE) ? HSE_VALUE : HSI_VALUE;
234 __pll_clk_in /= 1000000UL;
235
236 /* Configure de max value to pll_p parameter */
237 pll_parameters.p = 3; /* Bit value for request pll_p */
238
239 /* Determinate the minimun pll_p for request clock */
240 do {
241 uint8_t idx = pll_parameters.p;
242 /* fvco_out = pll_out * pll_p (pag. 226 - reference manual) */
243 fvco_out = pll_clk_out * pll_p[idx];
244 if (fvco_out > RCC_PLLN_MAX_VALUE) {
245 /* If fvco_out is major than FVCO_MAX decrease pll_p
246 to get a valid value for fvco_out freq */
247 pll_parameters.p -= 1;
248 } else {
249 /* Otherwise break the while routine */
250 break;
251 }
252 } while(pll_parameters.p > 0);
253 /* Recalculate fvco with the last pll_p value */
254 fvco_out = pll_clk_out * pll_p[pll_parameters.p];
255 /* Compute other parameters */
256 pll_parameters.m = __pll_clk_in / fvco_in;
257 pll_parameters.n = fvco_out / fvco_in;
258 pll_parameters.q = fvco_out / 48;
259 /* PLL_Q Must not be less than 2 */
260 if (pll_parameters.q < 2)
261 {
262 pll_parameters.q = 2;
263 }
264 /* Compute APB preescalers to get max bus speeds */
265 pll_parameters.ppre1 = pll_clk_out / 42; /* 42MHz in APB1 case */
266 pll_parameters.ppre2 = pll_clk_out / 84; /* 84MHz in APB2 case */
267
268 /* Update current bus speeds */
269 rcc_clocks_current.apb1 = (uint32_t) 42000000;
270 rcc_clocks_current.apb2 = (uint32_t) 84000000;
271}
