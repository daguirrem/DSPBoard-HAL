122void spi_gpio_config(spi_t *spi)
123{
124 /* Output configuration (MOSI & SCK) */
125 gpio_config_all(
126 spi->gpio_spi,
127 GPIO_CONFIG_MODE_ALTERNATE | GPIO_CONFIG_OUTPUT_PP | GPIO_CONFIG_SPEED_VHIGH,
128 spi->mosi_pin | spi->sck_pin
129 );
130
131 /* Input configuration (MISO) */
132 gpio_config_all(
133 spi->gpio_spi,
134 GPIO_CONFIG_MODE_ALTERNATE | GPIO_CONFIG_PULL_UP,
135 spi->miso_pin
136 );
137
138 /* Enable SPI alternate function at GPIO */
139 gpio_config_af(
140 spi->gpio_spi, GPIO_ALT_AF5,
141 spi->sck_pin | spi->miso_pin | spi->mosi_pin
142 );
143
144 /* Configure NSS */
145 gpio_io_t ss_pines = 0;
146 for (uint8_t i = 0; i < spi->ss_size; i++)
147 {
148 ss_pines |= spi->ss_pin[i];
149 }
150 if (spi->software_ss)
151 {
152 gpio_config_all(
153 spi->gpio_ss,
154 GPIO_CONFIG_MODE_OUTPUT | GPIO_CONFIG_SPEED_HIGH,
155 ss_pines
156 );
157 gpio_port_set(spi->gpio_ss, ss_pines);
158 }
159 else
160 {
161 // TODO: Hardware implementation
162 }
163}

165void spi_master_init(spi_t *spi)
166{
167 spi_gpio_config(spi);
168 spi->spi_p->CR1 =
169 spi->transfer_format | spi->data_format | spi->clk_pha | spi->clk_pol
170 | spi->mode |spi->clk_prescaler;
171
172 if(spi->software_ss)
173 {
174 /* If required enable software SS control */
175 spi->spi_p->CR1 |= SPI_CR1_SSM | SPI_CR1_SSI;
176 }
177 else
178 {
179 /* TODO: SS hardware configuration */
180 }
181 spi->spi_p->CR1 |= SPI_CR1_MSTR;
182
183 spi->spi_p->CR2 = 0;
184 spi->spi_p->CR1 |= SPI_CR1_SPE;
185
186 /* Compute spi clk */
187 const uint16_t lockup_prescaler[] = {2, 4, 8, 16, 32, 64, 128, 256};
188 const uint8_t i = spi->clk_prescaler >> SPI_CR1_BR_Pos;
189
190 float *clk = (float *) &(spi->clk);
191 *clk = rcc_clk_get_apb1() / lockup_prescaler[i];
192}

194void spi_master_write_raw(spi_t *spi, uint16_t data)
195{
196 /* Set data to transfer at DR register */
197 spi->spi_p->DR = data;
198 /* Wait until DR is empty & SPI transaction is done*/
199 while(!(spi->spi_p->SR & SPI_SR_TXE) || (spi->spi_p->SR & SPI_SR_BSY));
200}

202uint16_t spi_master_write_and_read_raw(spi_t *spi, uint16_t data)
203{
204 /* Set data to transfer at DR register */
205 spi->spi_p->DR = data;
206
207 /* Wait until:
208 * TX DR is empty
209 * RX DR is full
210 * SPI transaction is done.
211 */
212 while( !(spi->spi_p->SR & SPI_SR_TXE)
213 || !(spi->spi_p->SR & SPI_SR_RXNE) || (spi->spi_p->SR & SPI_SR_BSY));
214
215 /* Retrun readed data */
216 return (uint16_t) spi->spi_p->DR;
217}
