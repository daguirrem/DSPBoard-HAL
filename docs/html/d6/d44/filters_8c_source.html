151float dsp_filter_fir(fir_t *filter, filter_type_t type, float x_in)
152{
153 /* Function optimization */
154 #if defined(FIR_OPTIMICED) && (FIR_OPTIMICED==1)
155
156 /* Optimiced version use a do-while for x array shift & convolution
157 * (y = b*x).
158 * This version is apriximated 10us @168MHz faster than no optimiced version.
159 */
160
161 /* La versión optimizada hace un do-while para el corrimiento del arreglo
162 * de entrada (x) y la convolución (y=b*x).
163 * Esta versión es aproximadamente 10us más rápida que la versión no optimizada.
164 */
165
166 uint16_t s = filter->order; /* Array size */
167 float *x = filter->x; /* Initial address of x array (x[0]) */
168 float *_x = x + s; /* Final address of x array (x[s])*/
169 const float *b; /* Initial B coefficients array address (b[0]) */
170 const float *_b; /* Final address of B coefficientes array address (b[s]) */
171
172 switch (type)
173 {
174 case FILTER_TYPE_LOWPASS:
175 b = (filter->b.low);
176 break;
177 case FILTER_TYPE_HIGHPASS:
178 b = (filter->b.high);
179 break;
180 case FILTER_TYPE_BANDPASS:
181 b = (filter->b.pass);
182 break;
183 case FILTER_TYPE_BANDSTOP:
184 default:
185 b = (filter->b.stop);
186 break;
187 }
188
189 _b = b + s;
190
191 /* FIR: */
192 float y = 0;
193
194 do {
195 /* x array shift */
196 /* Where:
197 * x[1] = x[0], x[2] = x[1],...x[FIR_ORDER-1] = x[FIR_ORDER-2]
198 */
199 *_x = *(_x - 1);
200
201 /* Convolution */
202 /* Where:
203 * x[0] = x[n]; x[1] = x[n-1] ...
204 * y[n] = b[0]*x[n] + b[1]*x[n-1] + ... b[FIR_ORDER-1] * x[n - (FIR_ORDER-1)]
205 */
206 y += *(_x) * *(_b);
207 _x -=1; _b -= 1;
208 } while(--s);
209
210 /* x[0] = Input (Last entry) */
211 filter->x[0] = x_in;
212 y += b[0] * x[0];
213
214 #else /* Not optimiced version */
215
216 /* The no optimiced version use 2 independient for to shift and convolution.
217 * This version is aproximated 10us @168MHz slower than optimiced version.
218 */
219
220 /* La versión no optimizada hace uso de 2 "fors" independientes para el
221 * corrimiento y la convolución del filtro.
222 * Esta versión es aproximadamente 10us más lenta que la versión optimizada.
223 */
224
225 for (size_t k = filter->order; k > 0; k--)
226 {
227 /* x array shift */
228 /* Where:
229 * x[1] = x[0], x[2] = x[1],...x[FIR_ORDER-1] = x[FIR_ORDER-2]
230 */
231 filter->x[k] = filter->x[k-1];
232 }
233 /* x[0] = Input (Last entry) */
234 filter->x[0] = x_in;
235
236 /* Get a filter type to execute */
237 float *b;
238 switch (type)
239 {
240 case FILTER_TYPE_LOWPASS:
241 b = (filter->b.low);
242 break;
243 case FILTER_TYPE_HIGHPASS:
244 b = (filter->b.high);
245 break;
246 case FILTER_TYPE_BANDPASS:
247 b = (filter->b.pass);
248 break;
249 case FILTER_TYPE_BANDSTOP:
250 default:
251 b = (filter->b.stop);
252 break;
253 }
254
255 /* FIR */
256 /* Where:
257 * x[0] = x[n]; x[1] = x[n-1] ...
258 * y[n] = b[0]*x[n] + b[1]*x[n-1] + ... b[FIR_ORDER-1] * x[n - (FIR_ORDER-1)]
259 */
260 float y = 0; /* Acumulator */
261
262 for (size_t k = 0; k <= filter->order; k++)
263 {
264 /* Where x[0] = x[n]; x[1] = x[n-1] ... */
265 y += b[k] * filter->x[k];
266 }
267
268#endif /* !defined(FIR_OPTIMICED) || (FIR_OPTMICED==0)*/
269
270 /* Save y output on filter variable & return it */
271 filter->y = y;
272 return y;
273}

423float dsp_filter_iir(iir_t *filter, filter_type_t type, float x_in)
424{
425 for (size_t k = filter->order; k > 0; k--)
426 {
427 /* 'x' & 'y' arrays shift */
428 /* x[1] = x[0], x[2] = x[1],...x[IIR_ORDER] = x[IIR_ORDER-1] */
429 /* y[1] = y[0], y[2] = y[1],...y[IIR_ORDER] = y[IIR_ORDER-1] */
430 filter->x[k] = filter->x[k-1];
431 filter->y[k] = filter->y[k-1];
432 }
433 /* x[0] = Last input */
434 filter->x[0] = x_in;
435
436 const float *a;
437 const float *b;
438 switch (type)
439 {
440 case FILTER_TYPE_LOWPASS:
441 a = (filter->a.low);
442 b = (filter->b.low);
443 break;
444 case FILTER_TYPE_HIGHPASS:
445 a = (filter->a.high);
446 b = (filter->b.high);
447 break;
448 case FILTER_TYPE_BANDPASS:
449 a = (filter->a.pass);
450 b = (filter->b.pass);
451 break;
452 case FILTER_TYPE_BANDSTOP:
453 default:
454 a = (filter->a.stop);
455 b = (filter->b.stop);
456 break;
457 }
458 /* IIR: */
459 /* y[n] = b[0]*x[n] + b[1]*x[n-1] + ... b[IIR_ORDER] * x[n - (IIR_ORDER)]
460 * - a[1]*y[n-1] - a[2]*x[n-2] + ... a[IIR_ORDER] * y[n - (IIR_ORDER)]
461 */
462 float y = 0; /* Acumulator */
463 for (size_t k = 0; k <= filter->order; k++)
464 {
465 /* Donde x[0] = x[n]; x[1] = x[n-1] ... */
466 y += b[k] * filter->x[k];
467 }
468 for (size_t k = 1; k <= filter->order; k++)
469 {
470 /* Donde y[1] = y[n-1]; y[2] = y[n-2] ... */
471 y -= a[k] * filter->y[k];
472 }
473
474 /* Save y output on filter variable & return it */
475 filter->y[0] = y;
476 return y;
