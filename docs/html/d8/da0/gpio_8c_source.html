32/*
33 * Update each configuration register of masked IOs of GPIO in accordance of
34 * custom configuration mask.
35 */
36void gpio_config_all (GPIO_TypeDef *gpio, gpio_config_t config, gpio_io_t mask)
37{
38 /* Discriminate the CCCC bits of config mask [\config]*/
39 uint8_t configurator_enable = (config & 0xF00) >> 8;
40 /* MODER CONFIG = CCC1 */
41 if (configurator_enable & 0x1) {
42 /* Discriminate the MM bits of config mask */
43 uint8_t gpio_mode = (config & 0x3);
44 uint32_t _mask = (uint32_t) mask;
45 /* Enable desired gpio mode to each IO Modder configurator if IO is
46 masked */
47 for (uint8_t bit = 0; bit < 32; bit += 2) {
48 if (_mask & 0x1) {
49 gpio->MODER &= ~(0x3 << bit);
50 gpio->MODER |= (gpio_mode << bit);
51 }
52 _mask >>= 1;
53 }
54 }
55 /* OUTPUT MODE CONFIG = CC1C */
56 if (configurator_enable & 0x2) {
57 /* Pass through the O bits mask to Output Mode Register configurator */
58 uint16_t output_mode = (config & 0x4) >> 2;
59 gpio->OTYPER &= ~mask;
60 gpio->OTYPER |= output_mode ? mask : 0;
61 }
62 /* PULL MODE CONFIG = C1CC */
63 if (configurator_enable & 0x4) {
64 /* Discriminate the PP bits of config mask */
65 uint8_t pull_mode = (config & 0x18) >> 3;
66 uint32_t _mask = (uint32_t) mask;
67 /* Enable desired pull input mode to each IO Pull configurator if IO
68 is masked */
69 for (uint8_t bit = 0; bit < 32; bit += 2) {
70 if (_mask & 0x1) {
71 gpio->PUPDR &= ~(0x3 << bit);
72 gpio->PUPDR |= pull_mode << bit;
73 }
74 _mask >>= 1;
75 }
76 }
77 /* SPEED CONFIG = 1CCC */
78 if (configurator_enable & 0x8) {
79 /* Discriminate the SS bits of mask */
80 uint8_t speed = (config & 0x60) >> 5;
81 uint32_t _mask = (uint32_t) mask;
82 /* Enable desired speed mode to each IO Speed configurator if IO
83 is masked */
84 for (uint8_t bit = 0; bit < 32; bit += 2) {
85 if (_mask & 0x1) {
86 gpio->OSPEEDR &= ~(0x3 << bit);
87 gpio->OSPEEDR |= speed << bit;
88 }
89 _mask >>= 1;
90 }
91 }
92}
